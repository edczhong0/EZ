Schema Reference 
contacts  ( id INTEGER PK, fn TEXT, org TEXT )

phones    ( id INTEGER PK,
            contact_id → contacts.id,
            number TEXT,
            type   TEXT  DEFAULT 'VOICE' )

emails    ( id INTEGER PK,
            contact_id → contacts.id,
            address TEXT )

-- Example joins
SELECT c.fn, p.number
FROM contacts c JOIN phones p ON p.contact_id = c.id

SELECT c.fn, e.address
FROM contacts c JOIN emails e ON e.contact_id = c.id
